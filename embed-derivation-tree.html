Open standalone

If the app does not load, set the iframe src to the public URL where the derivation-tree app is hosted (for example "https://example.com/derivation-tree/").

try{ var script = document.currentScript || (function(){var s=document.getElementsByTagName('script');return s[s.length-1]})(); var dataSrc = script && script.getAttribute && script.getAttribute('data-derivation-src'); var iframe = document.getElementById('derivFrame'); if(dataSrc && iframe){ iframe.src = dataSrc; } // Open standalone link var open = document.getElementById('openStandalone'); open.addEventListener('click', function(e){e.preventDefault(); window.open(iframe.src || './', '_blank');}); }catch(e){console.warn(e)} })();
